/**
 * @param {string} s
 * @param {string} p
 * @return {boolean}
 */
var isMatch = function (s, p) {


    // p有无*.
    let isDot = false;
    for (let i = 0; i < p.length; i++) {
        if (p[i] == '*' || p[i] == '.') {
            isDot = true
        }
    }
    if (isDot) {
        let reg = new RegExp(p)
        return reg.test(s)
    }else{
        return s==p? true : false
    }


};
console.log(isMatch('aa','a*'))
